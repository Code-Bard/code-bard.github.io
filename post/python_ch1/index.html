题目：输入三角形的三边，输出周长和面积。

import math #内置模块 a = float(input(“输入边长a："))

b = float(input(“输入边长b："))

c = float(input(“输入边长c："))

h = (a+ b+ c)/2

s = math.sqrt(h*(h-a)*(h-b)*(h-c))

l = a+ b+ c

print(“三角形的周长："，l)

print(“三角形的面积："，s)
